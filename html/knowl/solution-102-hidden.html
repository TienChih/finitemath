\begin{align*} f'(x)\amp = \frac{d}{dx}[e^xx^2]\\ \amp =_{PR} \frac{d}{dx}[e^x]x^2+e^x\frac{d}{dx}[x^2]\\ \amp = e^xx^2+e^x2x\text{.} \end{align*}

\begin{align*} h'(x)\amp = \frac{d}{dx}[2^{\frac{x}{x+1}}]\\ \amp =_{CR} 2^{\frac{x}{x+1}}\ln(2)\frac{d}{dx}[\frac{x}{x+1}]\\ \amp =_{QR} 2^{\frac{x}{x+1}}\ln(2)\frac{(x+1)\frac{d}{dx}[x]-x\frac{d}{dx}[x+1]}{(x+1)^2}\\ \amp = \frac{2^{\frac{x}{x+1}}\ln(2)}{(x+1)^2} \end{align*}
