\begin{equation*} \begin{array}{|c|c|} \hline x \amp \text{Frequency}\\ \hline 1 \amp 1\\ 2 \amp 4\\ 3 \amp 6\\ 4 \amp 3\\ 5 \amp 2\\ 6 \amp 1\\ 7 \amp 1\\ \hline \end{array} \end{equation*}
