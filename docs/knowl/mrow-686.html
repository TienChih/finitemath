\begin{align*} \frac{d}{dx}[a^x]\amp = \frac{d}{dx}[(e^{\ln(a)})^x],\ \text{ since exponentiating by \(e\) and \(\ln\) are inverses }\\ \amp = \frac{d}{dx}[e^{\ln(a)x}]\\ \amp = e^{\ln(a)x}\frac{d}{dx}[\ln(a)x],\ \text{ by chain rule }\\ \amp = e^{\ln(a)x}\ln(a)\cdot1\\ \amp = a^x\ln(a) \end{align*}
