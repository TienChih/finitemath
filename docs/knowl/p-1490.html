\begin{equation*} \begin{pmatrix} 0.6 \amp 0.2 \\ 0.4 \amp 0.8\end{pmatrix}\begin{pmatrix} 800 \\ 1200 \end{pmatrix}=\begin{pmatrix} 720 \\ 1280 \end{pmatrix}. \end{equation*}

\begin{align*} \begin{pmatrix} 0.6 \amp 0.2 \\ 0.4 \amp 0.8\end{pmatrix}\begin{pmatrix} 800 \\ 1200 \end{pmatrix} \amp=\begin{pmatrix} 0.6 \amp 0.2 \\ 0.4 \amp 0.8\end{pmatrix}\left(\begin{pmatrix} 0.6 \amp 0.2 \\ 0.4 \amp 0.8\end{pmatrix}\begin{pmatrix} 1000 \\ 1000 \end{pmatrix}\right)\\ \amp=\begin{pmatrix} 0.6 \amp 0.2 \\ 0.4 \amp 0.8\end{pmatrix}^2\begin{pmatrix} 1000 \\ 1000 \end{pmatrix}\\ \amp=\begin{pmatrix} 0.44 \amp 0.28 \\ 0.56 \amp 0.72\end{pmatrix}\begin{pmatrix} 1000 \\ 1000 \end{pmatrix}\\ \amp=\begin{pmatrix} 720 \\ 1280 \end{pmatrix} \end{align*}
